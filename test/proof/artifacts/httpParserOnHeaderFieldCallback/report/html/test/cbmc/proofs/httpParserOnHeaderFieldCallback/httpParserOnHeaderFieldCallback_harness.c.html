25 /**
26 * @file httpParserOnHeaderFieldCallback_harness.c
27 * @brief Implements the proof harness for httpParserOnHeaderFieldCallback function.
28 */
29
30 #include "http_cbmc_state.h"
31 #include "llhttp.h"
32 #include "callback_stubs.h"
33
34 int __CPROVER_file_local_core_http_client_c_httpParserOnHeaderFieldCallback( llhttp_t * pHttpParser,
35 const char * pLoc,
36 size_t length );
37
38 void httpParserOnHeaderFieldCallback_harness()
39 {
40 llhttp_t * pHttpParser;
41 HTTPParsingContext_t * pParsingContext;
42 HTTPResponse_t * pResponse;
43 HTTPClient_ResponseHeaderParsingCallback_t headerParserCallback;
44 size_t length, locOffset;
45 char * pLoc;
46
47 pHttpParser = allocateHttpSendParser( NULL );
48
49 pParsingContext = ( HTTPParsingContext_t * ) ( pHttpParser->data );
50 headerParserCallback.onHeaderCallback = onHeaderCallbackStub;
51
52 pResponse = pParsingContext->pResponse;
53 pResponse->pHeaderParsingCallback = &headerParserCallback;
54
55 __CPROVER_assume( locOffset <= pResponse->bufferLen );
56 __CPROVER_assume( length <= pResponse->bufferLen - locOffset );
57 pLoc = pResponse->pBuffer + locOffset;
58
59 /* This assumption suppresses an overflow error when incrementing pResponse->headerCount. */
60 __CPROVER_assume( pResponse->headerCount < SIZE_MAX );
61
62 __CPROVER_file_local_core_http_client_c_httpParserOnHeaderFieldCallback( pHttpParser, pLoc, length );
63 }
