353 /**
354 * @ingroup http_struct_types
355 * @brief Represents header data that will be sent in an HTTP request.
356 *
357 * The memory for the header data buffer is supplied by the user. Information in
358 * the buffer will be filled by calling #HTTPClient_InitializeRequestHeaders and
359 * #HTTPClient_AddHeader. This buffer may be automatically filled with the
360 * Content-Length header in #HTTPClient_Send, please see
361 * HTTP_MAX_CONTENT_LENGTH_HEADER_LENGTH for the maximum amount of space needed
362 * to accommodate the Content-Length header.
363 */
364 typedef struct HTTPRequestHeaders
365 {
366 /**
367 * @brief Buffer to hold the raw HTTP request headers.
368 *
369 * This buffer is supplied by the application.
370 *
371 * This buffer is owned by the library during #HTTPClient_AddHeader,
372 * #HTTPClient_AddRangeHeader, #HTTPClient_InitializeRequestHeaders, and
373 * #HTTPClient_Send. This buffer should not be modified until
374 * after these functions return.
375 *
376 * For optimization this buffer may be re-used with the response. The user
377 * can re-use this buffer for the storing the response from the server in
378 * #HTTPResponse_t.pBuffer.
379 */
380 uint8_t * pBuffer;
381 size_t bufferLen; /**< The length of pBuffer in bytes. */
382
383 /**
384 * @brief The actual size in bytes of headers in the buffer. This field
385 * is updated by the HTTP Client library functions #HTTPClient_AddHeader,
386 * and #HTTPClient_InitializeRequestHeaders.
387 */
388 size_t headersLen;
389 } HTTPRequestHeaders_t;

586 /**
587 * @brief Initialize the request headers, stored in
588 * #HTTPRequestHeaders_t.pBuffer, with initial configurations from
589 * #HTTPRequestInfo_t. This method is expected to be called before sending a
590 * new request.
591 *
592 * Upon return, #HTTPRequestHeaders_t.headersLen will be updated with the number
593 * of bytes written.
594 *
595 * Each line in the header is listed below and written in this order:
596 * <#HTTPRequestInfo_t.pMethod> <#HTTPRequestInfo_t.pPath> <#HTTP_PROTOCOL_VERSION>
597 * User-Agent: <#HTTP_USER_AGENT_VALUE>
598 * Host: <#HTTPRequestInfo_t.pHost>
599 *
600 * Note that "Connection" header can be added and set to "keep-alive" by
601 * activating the HTTP_REQUEST_KEEP_ALIVE_FLAG in #HTTPRequestInfo_t.reqFlags.
602 *
603 * @param[in] pRequestHeaders Request header buffer information.
604 * @param[in] pRequestInfo Initial request header configurations.
605 * @return One of the following:
606 * - #HTTPSuccess (If successful)
607 * - #HTTPInvalidParameter (If any provided parameters or their members are invalid.)
608 * - #HTTPInsufficientMemory (If provided buffer size is not large enough to hold headers.)
609 *
610 * **Example**
611 * @code{c}
612 * HTTPStatus_t httpLibraryStatus = HTTPSuccess;
613 * // Declare an HTTPRequestHeaders_t and HTTPRequestInfo_t.
614 * HTTPRequestHeaders_t requestHeaders = { 0 };
615 * HTTPRequestInfo_t requestInfo = { 0 };
616 * // A buffer that will fit the Request-Line, the User-Agent header line, and
617 * // the Host header line.
618 * uint8_t requestHeaderBuffer[ 256 ] = { 0 };
619 *
620 * // Set a buffer to serialize request headers to.
621 * requestHeaders.pBuffer = requestHeaderBuffer;
622 * requestHeaders.bufferLen = 256;
623 *
624 * // Set the Method, Path, and Host in the HTTPRequestInfo_t.
625 * requestInfo.pMethod = HTTP_METHOD_GET;
626 * requestInfo.methodLen = sizeof( HTTP_METHOD_GET ) - 1U;
627 * requestInfo.pPath = "/html/rfc2616"
628 * requestInfo.pathLen = sizeof( "/html/rfc2616" ) - 1U;
629 * requestInfo.pHost = "tools.ietf.org"
630 * requestInfo.hostLen = sizeof( "tools.ietf.org" ) - 1U;
631 * requestInfo.reqFlags |= HTTP_REQUEST_KEEP_ALIVE_FLAG;
632 *
633 * httpLibraryStatus = HTTPClient_InitializeRequestHeaders( &requestHeaders,
634 * &requestInfo );
635 * @endcode
636 */
637 /* @[declare_httpclient_initializerequestheaders] */
638 HTTPStatus_t HTTPClient_InitializeRequestHeaders( HTTPRequestHeaders_t * pRequestHeaders,
639 const HTTPRequestInfo_t * pRequestInfo );
640 /* @[declare_httpclient_initializerequestheaders] */

642 /**
643 * @brief Add a header to the request headers stored in
644 * #HTTPRequestHeaders_t.pBuffer.
645 *
646 * Upon return, pRequestHeaders->headersLen will be updated with the number of
647 * bytes written.
648 *
649 * Headers are written in the following format:
650 *
651 * @code
652 * <field>: <value>\r\n\r\n
653 * @endcode
654 *
655 * The trailing `\r\n` that denotes the end of the header lines is overwritten,
656 * if it already exists in the buffer.
657 *
658 * @note This function validates only that `\r`, `\n`, and `:` are not present
659 * in @p pValue or @p pField. `:` is allowed in @p pValue.
660 *
661 * @param[in] pRequestHeaders Request header buffer information.
662 * @param[in] pField The header field name to write.
663 * The data should be ISO 8859-1 (Latin-1) encoded per the HTTP standard,
664 * but the API does not perform the character set validation.
665 * @param[in] fieldLen The byte length of the header field name.
666 * @param[in] pValue The header value to write.
667 * The data should be ISO 8859-1 (Latin-1) encoded per the HTTP standard,
668 * but the API does not perform the character set validation.
669 * @param[in] valueLen The byte length of the header field value.
670 *
671 * @return One of the following:
672 * - #HTTPSuccess (If successful.)
673 * - #HTTPInvalidParameter (If any provided parameters or their members are invalid.)
674 * - #HTTPInsufficientMemory (If application-provided buffer is not large enough to hold headers.)
675 * - #HTTPSecurityAlertInvalidCharacter (If an invalid character was found in @p pField or @p pValue.)
676 *
677 * **Example**
678 * @code{c}
679 * HTTPStatus_t httpLibraryStatus = HTTPSuccess;
680 * // Assume that requestHeaders has already been initialized with
681 * // HTTPClient_InitializeRequestHeaders().
682 * HTTPRequestHeaders_t requestHeaders;
683 *
684 * httpLibraryStatus = HTTPClient_AddHeader( &requestHeaders,
685 * "Request-Header-Field",
686 * sizeof( "Request-Header-Field" ) - 1U,
687 * "Request-Header-Value",
688 * sizeof("Request-Header-Value") - 1U );
689 * @endcode
690 */
691 /* @[declare_httpclient_addheader] */
692 HTTPStatus_t HTTPClient_AddHeader( HTTPRequestHeaders_t * pRequestHeaders,
693 const char * pField,
694 size_t fieldLen,
695 const char * pValue,
696 size_t valueLen );
697 /* @[declare_httpclient_addheader] */
698
699 /**
700 * @brief Add the byte range request header to the request headers store in
701 * #HTTPRequestHeaders_t.pBuffer.
702 *
703 * For example, if requesting for the first 1kB of a file the following would be
704 * written: `Range: bytes=0-1023\r\n\r\n`.
705 *
706 * The trailing `\r\n` that denotes the end of the header lines is overwritten,
707 * if it already exists in the buffer.
708 *
709 * There are 3 different forms of range specification, determined by the
710 * combination of @a rangeStartOrLastNBytes and @a rangeEnd parameter values:
711 *
712 * 1. Request containing both parameters for the byte range [rangeStart, rangeEnd]
713 * where @a rangeStartOrLastNBytes <= @a rangeEnd.
714 * Example request header line: `Range: bytes=0-1023\r\n` for requesting bytes in the range [0, 1023].<br>
715 * **Example**
716 * @code{c}
717 * HTTPStatus_t httpLibraryStatus = HTTPSuccess;
718 * // Assume that requestHeaders has already been initialized with
719 * // HTTPClient_InitializeRequestHeaders().
720 * HTTPRequestHeaders_t requestHeaders;
721 *
722 * // Request for bytes 0 to 1023.
723 * httpLibraryStatus = HTTPClient_AddRangeHeader( &requestHeaders, 0, 1023 );
724 * @endcode
725 *
726 * 2. Request for the last N bytes, represented by @p rangeStartOrlastNbytes.
727 * @p rangeStartOrlastNbytes should be negative and @p rangeEnd should be
728 * #HTTP_RANGE_REQUEST_END_OF_FILE.
729 * Example request header line: `Range: bytes=-512\r\n` for requesting the last 512 bytes
730 * (or bytes in the range [512, 1023] for a 1KB sized file).<br>
731 * **Example**
732 * @code{c}
733 * HTTPStatus_t httpLibraryStatus = HTTPSuccess;
734 * // Assume that requestHeaders has already been initialized with
735 * // HTTPClient_InitializeRequestHeaders().
736 * HTTPRequestHeaders_t requestHeaders;
737 *
738 * // Request for the last 512 bytes.
739 * httpLibraryStatus = HTTPClient_AddRangeHeader( &requestHeaders, -512, HTTP_RANGE_REQUEST_END_OF_FILE)
740 * @endcode
741 *
742 * 3. Request for all bytes (till the end of byte sequence) from byte N,
743 * represented by @p rangeStartOrlastNbytes.
744 * @p rangeStartOrlastNbytes should be >= 0 and @p rangeEnd should be
745 * #HTTP_RANGE_REQUEST_END_OF_FILE.<br>
746 * Example request header line: `Range: bytes=256-\r\n` for requesting all bytes after and
747 * including byte 256 (or bytes in the range [256,1023] for a 1kB sized file).<br>
748 * **Example**
749 * @code{c}
750 * HTTPStatus_t httpLibraryStatus = HTTPSuccess;
751 * // Assume that requestHeaders has already been initialized with
752 * // HTTPClient_InitializeRequestHeaders().
753 * HTTPRequestHeaders_t requestHeaders;
754 *
755 * // Request for all bytes from byte 256 onward.
756 * httpLibraryStatus = HTTPClient_AddRangeHeader( &requestHeaders, 256, HTTP_RANGE_REQUEST_END_OF_FILE)
757 * @endcode
758 *
759 * @param[in] pRequestHeaders Request header buffer information.
760 * @param[in] rangeStartOrlastNbytes Represents either the starting byte
761 * for a range OR the last N number of bytes in the requested file.
762 * @param[in] rangeEnd The ending range for the requested file. For end of file
763 * byte in Range Specifications 2. and 3., #HTTP_RANGE_REQUEST_END_OF_FILE
764 * should be passed.
765 *
766 * @return Returns the following status codes:
767 * #HTTPSuccess, if successful.
768 * #HTTPInvalidParameter, if input parameters are invalid, including when
769 * the @p rangeStartOrlastNbytes and @p rangeEnd parameter combination is invalid.
770 * #HTTPInsufficientMemory, if the passed #HTTPRequestHeaders_t.pBuffer
771 * contains insufficient remaining memory for storing the range request.
772 */
773 /* @[declare_httpclient_addrangeheader] */
774 HTTPStatus_t HTTPClient_AddRangeHeader( HTTPRequestHeaders_t * pRequestHeaders,
775 int32_t rangeStartOrlastNbytes,
776 int32_t rangeEnd );
777 /* @[declare_httpclient_addrangeheader] */

842 /**
843 * @brief Send the request headers in #HTTPRequestHeaders_t.pBuffer and request
844 * body in @p pRequestBodyBuf over the transport. The response is received in
845 * #HTTPResponse_t.pBuffer.
846 *
847 * If #HTTP_SEND_DISABLE_CONTENT_LENGTH_FLAG is not set in parameter @p sendFlags,
848 * then the Content-Length to be sent to the server is automatically written to
849 * @p pRequestHeaders. The Content-Length will not be written when there is
850 * no request body. If there is not enough room in the buffer to write the
851 * Content-Length then #HTTPInsufficientMemory is returned. Please see
852 * #HTTP_MAX_CONTENT_LENGTH_HEADER_LENGTH for the maximum Content-Length header
853 * field and value that could be written to the buffer.
854 *
855 * The application should close the connection with the server if any of the
856 * following errors are returned:
857 * - #HTTPSecurityAlertExtraneousResponseData
858 * - #HTTPSecurityAlertInvalidChunkHeader
859 * - #HTTPSecurityAlertInvalidProtocolVersion
860 * - #HTTPSecurityAlertInvalidStatusCode
861 * - #HTTPSecurityAlertInvalidCharacter
862 * - #HTTPSecurityAlertInvalidContentLength
863 *
864 * The @p pResponse returned is valid only if this function returns HTTPSuccess.
865 *
866 * @param[in] pTransport Transport interface, see #TransportInterface_t for
867 * more information.
868 * @param[in] pRequestHeaders Request configuration containing the buffer of
869 * headers to send.
870 * @param[in] pRequestBodyBuf Optional Request entity body. Set to NULL if there
871 * is no request body.
872 * @param[in] reqBodyBufLen The length of the request entity in bytes.
873 * @param[in] pResponse The response message and some notable response
874 * parameters will be returned here on success.
875 * @param[in] sendFlags Flags which modify the behavior of this function. Please
876 * see @ref http_send_flags for more information.
877 *
878 * @return One of the following:
879 * - #HTTPSuccess (If successful.)
880 * - #HTTPInvalidParameter (If any provided parameters or their members are invalid.)
881 * - #HTTPNetworkError (Errors in sending or receiving over the transport interface.)
882 * - #HTTPPartialResponse (Part of an HTTP response was received in a partially filled response buffer.)
883 * - #HTTPNoResponse (No data was received from the transport interface.)
884 * - #HTTPInsufficientMemory (The response received could not fit into the response buffer
885 * or extra headers could not be sent in the request.)
886 * - #HTTPParserInternalError (Internal parsing error.)\n\n
887 * Security alerts are listed below, please see #HTTPStatus_t for more information:
888 * - #HTTPSecurityAlertExtraneousResponseData
889 * - #HTTPSecurityAlertInvalidChunkHeader
890 * - #HTTPSecurityAlertInvalidProtocolVersion
891 * - #HTTPSecurityAlertInvalidStatusCode
892 * - #HTTPSecurityAlertInvalidCharacter
893 * - #HTTPSecurityAlertInvalidContentLength
894 *
895 * **Example**
896 * @code{c}
897 * // Variables used in this example.
898 * HTTPStatus_t httpLibraryStatus = HTTPSuccess;
899 * TransportInterface_t transportInterface = { 0 };
900 * HTTPResponse_t = { 0 };
901 * char requestBody[] = "This is an example request body.";
902 *
903 * // Assume that requestHeaders has been initialized with
904 * // HTTPClient_InitializeResponseHeaders() and any additional headers have
905 * // been added with HTTPClient_AddHeader().
906 * HTTPRequestHeaders_t requestHeaders;
907 *
908 * // Set the transport interface with platform specific functions that are
909 * // assumed to be implemented elsewhere.
910 * transportInterface.recv = myPlatformTransportReceive;
911 * transportInterface.send = myPlatformTransportSend;
912 * transportInterface.pNetworkContext = myPlatformNetworkContext;
913 *
914 * // Set the buffer to receive the HTTP response message into. The buffer is
915 * // dynamically allocated for demonstration purposes only.
916 * response.pBuffer = ( uint8_t* )malloc( 1024 );
917 * response.bufferLen = 1024;
918 *
919 * httpLibraryStatus = HTTPClient_Send( &transportInterface,
920 * &requestHeaders,
921 * requestBody,
922 * sizeof( requestBody ) - 1U,
923 * &response,
924 * 0 );
925 *
926 * if( httpLibraryStatus == HTTPSuccess )
927 * {
928 * if( response.status == 200 )
929 * {
930 * // Handle a response Status-Code of 200 OK.
931 * }
932 * }
933 * @endcode
934 */
935 /* @[declare_httpclient_send] */
936 HTTPStatus_t HTTPClient_Send( const TransportInterface_t * pTransport,
937 HTTPRequestHeaders_t * pRequestHeaders,
938 const uint8_t * pRequestBodyBuf,
939 size_t reqBodyBufLen,
940 HTTPResponse_t * pResponse,
941 uint32_t sendFlags );

965 /**
966 * @brief Read a header from a buffer containing a complete HTTP response.
967 * This will return the location of the response header value in the
968 * #HTTPResponse_t.pBuffer buffer.
969 *
970 * The location, within #HTTPResponse_t.pBuffer, of the value found, will be
971 * returned in @p pValue. If the header value is empty for the found @p pField,
972 * then this function will return #HTTPSuccess, and set the values for
973 * @p pValueLoc and @p pValueLen as NULL and zero respectively. According to
974 * RFC 2616, it is not invalid to have an empty value for some header fields.
975 *
976 * @note This function should only be called on a complete HTTP response. If the
977 * request is sent through the #HTTPClient_Send function, the #HTTPResponse_t is
978 * incomplete until #HTTPClient_Send returns.
979 *
980 * @param[in] pResponse The buffer containing the completed HTTP response.
981 * @param[in] pField The header field name to read.
982 * @param[in] fieldLen The length of the header field name in bytes.
983 * @param[out] pValueLoc This will be populated with the location of the
984 * header value in the response buffer, #HTTPResponse_t.pBuffer.
985 * @param[out] pValueLen This will be populated with the length of the
986 * header value in bytes.
987 *
988 * @return One of the following:
989 * - #HTTPSuccess (If successful.)
990 * - #HTTPInvalidParameter (If any provided parameters or their members are invalid.)
991 * - #HTTPHeaderNotFound (Header is not found in the passed response buffer.)
992 * - #HTTPInvalidResponse (Provided response is not a valid HTTP response for parsing.)
993 * - #HTTPParserInternalError(If an error in the response parser.)
994 *
995 * **Example**
996 * @code{c}
997 * HTTPStatus_t httpLibraryStatus = HTTPSuccess;
998 * // Assume that response is returned from a successful invocation of
999 * // HTTPClient_Send().
1000 * HTTPResponse_t response;
1001 *
1002 * char * pDateLoc = NULL;
1003 * size_t dateLen = 0;
1004 * // Search for a "Date" header field. pDateLoc will be the location of the
1005 * // Date header value in response.pBuffer.
1006 * httpLibraryStatus = HTTPClient_ReadHeader( &response,
1007 * "Date",
1008 * sizeof("Date") - 1,
1009 * &pDateLoc,
1010 * &dateLen );
1011 * @endcode
1012 */
1013 /* @[declare_httpclient_readheader] */
1014 HTTPStatus_t HTTPClient_ReadHeader( const HTTPResponse_t * pResponse,
1015 const char * pField,
1016 size_t fieldLen,
1017 const char ** pValueLoc,
1018 size_t * pValueLen );
