25 /**
26 * @file HTTPClient_ReadHeader_llhttp_execute.c
27 * @brief A stub function for llhttp_execute for coverage of
28 * #HTTPClient_ReadHeader.
29 */
30
31 #include <stdbool.h>
32 #include <stdint.h>
33
34 #include "core_http_client.h"
35 #include "core_http_client_private.h"
36 #include "llhttp.h"
37
38 /**
39 * @brief Attains coverage when a variable needs to possibly contain two values.
40 */
41 bool nondet_bool();
42
43 llhttp_errno_t llhttp_execute( llhttp_t * parser,
44 const char * data,
45 size_t len )
46 {
47 char * pValue;
48 size_t fieldLength, fieldOffset, valueLength, valueOffset;
49 int http_errno;
50 findHeaderContext_t * pParsingContext;
51
52 __CPROVER_assert( parser != NULL,
53 "llhttp_execute parser is NULL" );
54 __CPROVER_assert( data != NULL,
55 "llhttp_execute data is NULL" );
56 __CPROVER_assert( len < CBMC_MAX_OBJECT_SIZE,
57 "llhttp_execute len >= CBMC_MAX_OBJECT_SIZE" );
58
59 parser->error = http_errno;
60
61 __CPROVER_assume( fieldLength <= len );
62 __CPROVER_assume( fieldOffset < fieldLength );
63 __CPROVER_assume( valueLength <= len );
64 __CPROVER_assume( valueOffset < valueLength );
65
66 pParsingContext = ( findHeaderContext_t * ) ( parser->data );
67 pParsingContext->pField = data + fieldOffset;
68 pParsingContext->fieldLen = fieldLength;
69 pParsingContext->pValueLoc = NULL;
70 pParsingContext->pValueLen = 0;
71 pParsingContext->fieldFound = nondet_bool() ? 0 : 1;
72
73 if( pParsingContext->fieldFound )
74 {
75 pParsingContext->valueFound = nondet_bool() ? 0 : 1;
76 }
77 else
78 {
79 pParsingContext->valueFound = 0;
80 }
81
82 if( pParsingContext->valueFound )
83 {
84 pValue = data + valueOffset;
85 pParsingContext->pValueLen = &valueLength;
86 pParsingContext->pValueLoc = &pValue;
87 }
88
89 return parser->error;
90 }
