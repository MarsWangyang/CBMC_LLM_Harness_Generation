47 void HTTPClient_AddRangeHeader_harness()
48 {
49 HTTPRequestHeaders_t * pRequestHeaders;
50 int32_t rangeStartOrlastNbytes;
51 int32_t rangeEnd;
52
53 /* Initialize and make assumptions for request headers. */
54 pRequestHeaders = allocateHttpRequestHeaders( NULL );
55 __CPROVER_assume( isValidHttpRequestHeaders( pRequestHeaders ) );
56
57 HTTPClient_AddRangeHeader( pRequestHeaders, rangeStartOrlastNbytes, rangeEnd );
58 }
