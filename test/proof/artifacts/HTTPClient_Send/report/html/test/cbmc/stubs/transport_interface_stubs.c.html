27 int32_t TransportInterfaceSendStub( NetworkContext_t * pNetworkContext,
28 void * pBuffer,
29 size_t bytesToSend )
30 {
31 /* The number of tries to send the message before this invocation */
32 static int32_t tries = 0;
33 /* The number of bytes considered sent after this invocation */
34 int32_t ret;
35
36 __CPROVER_assert( pBuffer != NULL,
37 "TransportInterfaceSendStub pBuffer is NULL" );
38
39 /****************************************************************
40 * The send method sends some portion of the message and returns the
41 * total number of bytes in the prefix sent so far. The send method
42 * is used in a loop of the form
43 *
44 * while ( send( conn, msg, len ) < len ) { ... }
45 *
46 * We need to bound the number of loop iterations, so we need to
47 * bound the number of times it takes for send to finish sending the
48 * message. We use a static variable 'tries' to count the number of
49 * times send has tried to send the message, and we force send to
50 * finish the message after MAX_TRIES tries.
51 ****************************************************************/
52
53 if( bytesToSend <= INT32_MAX )
54 {
55 __CPROVER_assume( ret <= ( int32_t ) bytesToSend );
56 }
57
58 tries++;
59
60 if( tries >= MAX_TRIES )
61 {
62 tries = 0;
63
64 /* In order to stop the looping on send we must return an error or
65 * bytesToSend. We return an error instead of bytesToSend because
66 * bytesToSend may be a value larger than INT32_MAX. */
67 ret = -1;
68 }
69
70 return ret;
71 }
