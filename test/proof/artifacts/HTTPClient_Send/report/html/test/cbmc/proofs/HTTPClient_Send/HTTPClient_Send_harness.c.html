69 void HTTPClient_Send_harness()
70 {
71 HTTPRequestHeaders_t * pRequestHeaders;
72 HTTPResponse_t * pResponse;
73 TransportInterface_t * pTransportInterface;
74 uint8_t * pRequestBodyBuf;
75 size_t reqBodyBufLen;
76 uint32_t sendFlags;
77
78 /* Initialize and make assumptions for request headers. */
79 pRequestHeaders = allocateHttpRequestHeaders( NULL );
80 __CPROVER_assume( isValidHttpRequestHeaders( pRequestHeaders ) );
81
82 /* Initialize and make assumptions for buffer to receive request body. */
83 __CPROVER_assume( reqBodyBufLen < CBMC_MAX_OBJECT_SIZE );
84 pRequestBodyBuf = mallocCanFail( reqBodyBufLen );
85
86 /* Initialize and make assumptions for response object. */
87 pResponse = allocateHttpResponse( NULL );
88 __CPROVER_assume( isValidHttpResponse( pResponse ) );
89
90 /* Initialize transport interface. */
91 pTransportInterface = allocateTransportInterface( NULL );
92
93 if( pTransportInterface != NULL )
94 {
95 /* Ideally, we want to set the function pointers below with __CPROVER_assume()
96 * but doing so makes CBMC run out of memory. */
97 pTransportInterface->send = nondet_bool() ? NULL : TransportInterfaceSendStub;
98 pTransportInterface->recv = nondet_bool() ? NULL : TransportInterfaceReceiveStub;
99 }
100
101 if( pResponse != NULL )
102 {
103 pResponse->getTime = nondet_boot() ? NULL : GetCurrentTimeStub;
104 }
105
106 HTTPClient_Send( pTransportInterface,
107 pRequestHeaders,
108 pRequestBodyBuf,
109 reqBodyBufLen,
110 pResponse,
111 sendFlags );
112 }
