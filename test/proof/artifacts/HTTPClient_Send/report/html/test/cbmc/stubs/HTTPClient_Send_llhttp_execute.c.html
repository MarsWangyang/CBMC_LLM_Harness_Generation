25 /**
26 * @file HTTPClient_Send_llhttp_execute.c
27 * @brief A stub function for llhttp_execute for coverage of
28 * #HTTPClient_Send.
29 */
30
31 #include <stdbool.h>
32 #include <stdint.h>
33
34 #include "core_http_client.h"
35 #include "core_http_client_private.h"
36 #include "llhttp.h"
37
38 /**
39 * @brief Attains coverage when a variable needs to possibly contain two values.
40 */
41 bool nondet_bool();
42
43 llhttp_errno_t llhttp_execute( llhttp_t * parser,
44 const char * data,
45 size_t len )
46 {
47 size_t fieldLength, valueLength;
48 HTTPParsingContext_t * pParsingContext;
49 int http_errno;
50
51 __CPROVER_assert( parser != NULL,
52 "llhttp_execute parser is NULL" );
53 __CPROVER_assert( data != NULL,
54 "llhttp_execute data is NULL" );
55 __CPROVER_assert( len < CBMC_MAX_OBJECT_SIZE,
56 "llhttp_execute len >= CBMC_MAX_OBJECT_SIZE" );
57
58 __CPROVER_assume( fieldLength <= len );
59 __CPROVER_assume( valueLength <= len );
60
61 parser->error = http_errno;
62
63 pParsingContext = ( HTTPParsingContext_t * ) ( parser->data );
64 /* Choose whether the parser found the header */
65 pParsingContext->pLastHeaderField = malloc( fieldLength );
66 __CPROVER_assume( pParsingContext->pLastHeaderField != NULL );
67 pParsingContext->state = HTTP_PARSING_COMPLETE;
68
69 if( pParsingContext->pLastHeaderField )
70 {
71 pParsingContext->lastHeaderFieldLen = fieldLength;
72 pParsingContext->pLastHeaderValue = malloc( valueLength );
73 __CPROVER_assume( pParsingContext->pLastHeaderValue != NULL );
74 pParsingContext->lastHeaderValueLen = valueLength;
75 }
76 else
77 {
78 pParsingContext->lastHeaderFieldLen = 0U;
79 pParsingContext->pLastHeaderValue = NULL;
80 pParsingContext->lastHeaderValueLen = 0U;
81 }
82
83 /* The body pointer is set by the httpParserOnBodyCallback. But since we are
84 * removing that from CBMC proof execution, the body has to be set here. */
85 size_t bodyOffset;
86
87 if( pParsingContext->pResponse->bufferLen == 0U )
88 {
89 bodyOffset = 0U;
90 }
91 else
92 {
93 /* Body offset can be anything as long as it doesn't exceed the buffer length
94 * and the length of the current data packet. */
95 __CPROVER_assume( bodyOffset < pParsingContext->pResponse->bufferLen );
96 __CPROVER_assume( bodyOffset < len );
97 }
98
99 pParsingContext->pResponse->pBody = pParsingContext->pBufferCur + bodyOffset;
100
101 if( parser->error == HPE_PAUSED )
102 {
103 /* When the parser is paused ensure that the error_pos member points to
104 * a valid location in the response buffer. */
105 size_t errorPosOffset;
106
107 __CPROVER_assume( errorPosOffset < pParsingContext->pResponse->bufferLen );
108 __CPROVER_assume( errorPosOffset < len );
109
110 parser->error_pos = pParsingContext->pResponse->pBuffer + errorPosOffset;
111 }
112
113 return parser->error;
114 }
