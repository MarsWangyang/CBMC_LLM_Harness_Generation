25 /**
26 * @file get_time_stub.c
27 * @brief A stub to mock the retrieval of current time.
28 */
29 #include "stdint.h"
30 #include "get_time_stub.h"
31
32
33 uint32_t GetCurrentTimeStub( void )
34 {
35 /* The HTTP relies on this timestamp in order to complete the network send
36 * and receive loops. Returning an unbounded timestamp does not add value to
37 * the proofs as the HTTP library uses this timestamp only in an arithmetic
38 * operation to get the difference. In C, arithmetic operations on unsigned
39 * integers are guaranteed to reliably wrap around with no adverse side
40 * effects. If the time returned was unbounded, the network send and receive
41 * loops could be unwound a large number of times making the proof execution
42 * very long. */
43 static uint32_t globalEntryTime = 0;
44
45 return globalEntryTime++;
46 }
