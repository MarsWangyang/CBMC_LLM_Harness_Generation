31 #ifndef TRANSPORT_INTERFACE_STUBS_H_
32 #define TRANSPORT_INTERFACE_STUBS_H_
33
34 #include <stdint.h>
35
36 #include "core_http_client.h"
37
38 #ifndef MAX_TRIES
39 #define MAX_TRIES 5
40 #endif

42 /**
43 * @brief Transport interface stub for mocking data sent over the network.
44 *
45 * @param[in] pNetworkContext Implementation-defined network context.
46 * @param[in] pBuffer Buffer containing the bytes to send over the network stack.
47 * @param[in] bytesToSend Number of bytes to send over the network.
48 *
49 * @return The number of bytes sent or a negative error code.
50 */
51 int32_t TransportInterfaceSendStub( NetworkContext_t * pNetworkContext,
52 void * pBuffer,
53 size_t bytesToSend );
54
55 /**
56 * @brief Transport interface for mocking data received over the network.
57 *
58 * @param[in] pNetworkContext Implementation-defined network context.
59 * @param[in] pBuffer Buffer to receive the data into.
60 * @param[in] bytesToRecv Number of bytes requested from the network.
61 *
62 * @return The number of bytes received or a negative error code.
63 */
64 int32_t TransportInterfaceReceiveStub( NetworkContext_t * pNetworkContext,
65 void * pBuffer,
66 size_t bytesToRecv );
67
68 #endif /* ifndef TRANSPORT_INTERFACE_STUBS_H_ */
