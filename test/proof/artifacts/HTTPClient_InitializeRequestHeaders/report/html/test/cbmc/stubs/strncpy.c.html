25 /**
26 * @file strncpy.c
27 * @brief Creates a stub for strncpy so that the proofs for HTTPClient_AddHeader,
28 * HTTPClient_AddRangeHeader, and HTTPClient_InitializeRequestHeaders, run much
29 * faster. This stub checks if, for the input copy length, the destination and
30 * source are valid accessible memory.
31 */
32
33 #include <string.h>
34
35 /* This is a clang macro not available on linux */
36 #ifndef __has_builtin
37 #define __has_builtin( x ) 0
38 #endif
39
40 #if __has_builtin( __builtin___strncpy_chk )
41 char * __builtin___strncpy_chk( char * dest,
42 const char * src,
43 size_t n,
44 size_t os )
45 {
46 __CPROVER_assert( __CPROVER_w_ok( dest, n ), "write" );
47 __CPROVER_assert( __CPROVER_r_ok( src, n ), "read" );
48 return dest;
49 }
50 #else
51 char * strncpy( char * dest,
52 const char * src,
53 size_t n )
54 {
55 __CPROVER_assert( __CPROVER_w_ok( dest, n ), "write" );
56 __CPROVER_assert( __CPROVER_r_ok( src, n ), "read" );
57 return dest;
58 }
59 #endif /* if __has_builtin( __builtin___strncpy_chk ) */
