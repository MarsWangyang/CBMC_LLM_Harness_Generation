25 /**
26 * @file HTTPClient_InitializeRequestHeaders_harness.c
27 * @brief Implements the proof harness for HTTPClient_InitializeRequestHeaders function.
28 */
29
30 #include "core_http_client.h"
31
32 #include "http_cbmc_state.h"
33
34 void HTTPClient_InitializeRequestHeaders_harness()
35 {
36 HTTPRequestHeaders_t * pRequestHeaders;
37 HTTPRequestInfo_t * pRequestInfo;
38
39 /* Initialize and make assumptions for the request headers object. */
40 pRequestHeaders = allocateHttpRequestHeaders( NULL );
41 __CPROVER_assume( isValidHttpRequestHeaders( pRequestHeaders ) );
42
43 /* Initialize and make assumptions for the request info object. */
44 pRequestInfo = allocateHttpRequestInfo( NULL );
45 __CPROVER_assume( isValidHttpRequestInfo( pRequestInfo ) );
46
47 HTTPClient_InitializeRequestHeaders( pRequestHeaders, pRequestInfo );
48 }
