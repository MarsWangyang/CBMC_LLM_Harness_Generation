190 /**
191 * @brief The minimum request-line in the headers has a possible one character
192 * custom method and a single forward / or asterisk * for the path:
193 *
194 * @code
195 * <1 character custom method> <1 character / or *> HTTP/1.x\r\n\r\n
196 * @endcode
197 *
198 * Therefore the minimum length is 16. If this minimum request-line is not
199 * satisfied, then the request headers to send are invalid.
200 *
201 * Note that custom methods are allowed per:
202 * https://tools.ietf.org/html/rfc2616#section-5.1.1.
203 */
204 #define HTTP_MINIMUM_REQUEST_LINE_LENGTH 16u

206 /**
207 * @brief The state of the response message parsed after function
208 * #parseHttpResponse returns.
209 */
210 typedef enum HTTPParsingState
211 {
212 HTTP_PARSING_NONE = 0, /**< The parser has not started reading any response. */
213 HTTP_PARSING_INCOMPLETE, /**< The parser found a partial response. */
214 HTTP_PARSING_COMPLETE /**< The parser found the entire response. */
215 } HTTPParsingState_t;

232 /**
233 * @brief The HTTP response parsing context for a response fresh from the
234 * server. This context is passed into the http-parser registered callbacks.
235 * The registered callbacks are private functions of the form
236 * httpParserXXXXCallbacks().
237 *
238 * The transitions of the httpParserXXXXCallback() functions are shown below.
239 * The XXXX is replaced by the strings in the state boxes:
240 *
241 * +---------------------+
242 * |onMessageBegin |
243 * +--------+------------+
244 * |
245 * |
246 * |
247 * v
248 * +--------+------------+
249 * |onStatus |
250 * +--------+------------+
251 * |
252 * |
253 * |
254 * v
255 * +--------+------------+
256 * |onStatusComplete |
257 * +--------+------------+
258 * |
259 * |
260 * |
261 * v
262 * +--------+------------+
263 * |onHeaderField +<---+
264 * +--------+------------+ |
265 * | |
266 * | |(More headers)
267 * | |
268 * v |
269 * +--------+------------+ |
270 * |onHeaderValue +----^
271 * +--------+------------+
272 * |
273 * |
274 * |
275 * v
276 * +--------+------------+
277 * |onHeadersComplete |
278 * +---------------------+
279 * |
280 * |
281 * |
282 * v
283 * +--------+------------+
284 * |onBody +<---+
285 * +--------+--------+---+ |
286 * | | |(Transfer-encoding chunked body)
287 * | | |
288 * | +--------+
289 * |
290 * v
291 * +--------+------------+
292 * |onMessageComplete |
293 * +---------------------+
294 */
295 typedef struct HTTPParsingContext
296 {
297 llhttp_t llhttpParser; /**< Third-party llhttp context. */
298 llhttp_settings_t llhttpSettings; /**< Third-party parser settings. */
299 HTTPParsingState_t state; /**< The current state of the HTTP response parsed. */
300 HTTPResponse_t * pResponse; /**< HTTP response associated with this parsing context. */
301 uint8_t isHeadResponse; /**< HTTP response is for a HEAD request. */
302
303 const char * pBufferCur; /**< The current location of the parser in the response buffer. */
304 const char * pLastHeaderField; /**< Holds the last part of the header field parsed. */
305 size_t lastHeaderFieldLen; /**< The length of the last header field parsed. */
306 const char * pLastHeaderValue; /**< Holds the last part of the header value parsed. */
307 size_t lastHeaderValueLen; /**< The length of the last value field parsed. */
308 } HTTPParsingContext_t;
