25 /**
26 * @file httpParserOnHeadersCompleteCallback_harness.c
27 * @brief Implements the proof harness for httpParserOnHeadersCompleteCallback function.
28 */
29
30 #include "http_cbmc_state.h"
31 #include "llhttp.h"
32 #include "callback_stubs.h"
33
34 int __CPROVER_file_local_core_http_client_c_httpParserOnHeadersCompleteCallback( llhttp_t * pHttpParser );
35
36 void httpParserOnHeadersCompleteCallback_harness()
37 {
38 llhttp_t * pHttpParser;
39 HTTPParsingContext_t * pParsingContext;
40 HTTPResponse_t * pResponse;
41 HTTPClient_ResponseHeaderParsingCallback_t headerParserCallback;
42 size_t bufferOffset;
43 size_t headerOffSet;
44
45 pHttpParser = allocateHttpSendParser( NULL );
46
47 pParsingContext = ( HTTPParsingContext_t * ) ( pHttpParser->data );
48 headerParserCallback.onHeaderCallback = onHeaderCallbackStub;
49
50 pResponse = pParsingContext->pResponse;
51 pResponse->pHeaderParsingCallback = &headerParserCallback;
52
53 /* The next place to parse must be between the start of the buffer and the
54 * end of the buffer. This assumption is an assertion in the function. */
55 __CPROVER_assume( bufferOffset < pResponse->bufferLen );
56 pParsingContext->pBufferCur = pResponse->pBuffer + bufferOffset;
57
58 /* The response headers MUST have been set prior or be NULL. This assumption
59 * is an assertion in the function. */
60 if( pResponse->pHeaders != NULL )
61 {
62 __CPROVER_assume( headerOffSet < bufferOffset );
63 pResponse->pHeaders = pResponse->pBuffer + headerOffSet;
64 }
65
66 /* This assumption suppresses an overflow error when incrementing pResponse->headerCount. */
67 __CPROVER_assume( pResponse->headerCount < SIZE_MAX );
68
69 __CPROVER_file_local_core_http_client_c_httpParserOnHeadersCompleteCallback( pHttpParser );
70 }
